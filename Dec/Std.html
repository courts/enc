def Std::hex(data)
  res = ""
  data = data.split("")
  until data.empty?
    res << data.shift(2).join().hex.chr
  end
  res
end

def Std::url_raw(data)
  data = data.split("%")
  data[1..-1].pack("H2"*(data.length - 1))
end
