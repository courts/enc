def HTML::dec(data)
  data.unpack("c*").map do |x|
    "&##{x};"
  end.join
end

def HTML::hex(data)
  data.split("").map do |x|
    "&#x" + x.unpack('H*')[0].upcase + ";"
  end.join
end
