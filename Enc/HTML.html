def HTML::dec(data)
  Std::ord_as_ary(data).map do |x|
    "&##{x};"
  end.join
end

def HTML::umlauts(data)
  umlauts = {
    'ä' => '&auml;',
    'ö' => '&ouml;',
    'ü' => '&uuml;',
    'Ä' => '&Auml;',
    'Ö' => '&Ouml;',
    'Ü' => '&Uuml;',
    'ß' => '&szlig;'
  }
  umlauts.each do |umlaut, repr|
    data.gsub!(umlaut, repr)
  end
  data
end
