def Java::bytearray(data)
  bytes = Std::hex_as_ary(data)
  java_ary = bytes.map {|x| "(byte) 0x#{x}"}
  ret = "byte data[] = new byte[]\n{"
  until java_ary.empty?
    ret << "\n   " + java_ary.shift(8).join(", ") + ","
  end
  ret << "\n};\n"
  return ret
end
