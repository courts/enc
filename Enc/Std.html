def Std::rand_upcase(data)
  data.split("").map do |x|
    if rand(2) == 0 then x.upcase else x end
  end.join
end

def Std::url(data, all=false)
  if all
    data.split("").map do |x|
      "%" + x.unpack('H*')[0]
    end.join.upcase
  else
    CGI.escape(data)
  end
end

def Std::url_raw(data)
  res = ""
  data.each_byte do |x|
    res << ("%" + (sprintf "%02X" % x))
  end
  res.upcase
end

def Std::xor(data, key)
  data.length.times {|x| data[x] = (data[x].ord ^ (key[x % key.length]).ord).chr }
  return data
end
