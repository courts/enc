def UTF8::bin(data)

    "\\x" + x.unpack('H*')[0].upcase
  end.join
end

def UTF8::utf8(data)
  data.split("").map do |x|
    val = x.unpack('H*')[0].upcase
    if val.length == 2
      "\\u00" + val
    else
      "\\u" + val
    end
  end.join
end
